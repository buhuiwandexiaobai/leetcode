#### 题目：判断一棵二叉树是否是对称二叉树。

#### 思路：对称二叉树就是本身和它的镜像相等。根据这一思路我们可以利用递归判断。另外我们还可以利用按层遍历的思想，对称二叉树的按层遍历时，先访问左子树再访问右子树应该和先访问右子树再访问左子树的顺序相同。
####

代码：
```
/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * } 
 */
/**
 * @param {TreeNode} root
 * @return {boolean}
 */
var isSymmetric = function(root) {
    var core = function(r1, r2) {
        if(r1 === null && r2 === null) {
            return true;
        }
        if(r1 === null || r2 === null || r1.val !== r2.val) {
            return false;
        }
        return core(r1.left, r2.right) && core(r1.right, r2.left);
        
    }
    return core(root, root);
};
var isSymmetric = function(root) {
    var queue1 = [root],
        queue2 = [root];
    while(queue1.length) {
        let r1 = queue1.pop(),
            r2 = queue2.pop();
        if(r1 === null && r2 === null) {
           continue;
        }
        if(r1 === null || r2 === null || r1.val !== r2.val) {
            return false;
        }
        queue1.push(r1.left);
        queue1.push(r1.right);
        queue2.push(r2.right);
        queue2.push(r2.left);
    }
    return true;
};
```
